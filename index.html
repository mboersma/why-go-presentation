package main

import (
	"fmt"
	"os"
	"strings"
)
func main() {
	args := os.Args[1:]
	if len(args) == 0 {
		fmt.Println("usage: hello [name]")
		os.Exit(1)
	}
	name := strings.Join(args, " ")
	fmt.Printf("Hello %s!\n", name)
}
